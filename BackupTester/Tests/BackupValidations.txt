module encryption and compression consistency - ordinary archive

test initial ordinary
date 2014-01-01
mkdir source
mkdir archive
call backup -date %DATE% backup source archive

list-verify .
.                                         -----D archive\
.                                         -----D archive\2014-01-01T00+00+00\
.                                         -----D source\
endlist

test try compressed
date + 1
call backup -date %DATE% -compress backup source archive

.

.Previous backups have compression setting incompatible with current setting
endoutput

.                                         -----D source\
endlist
